```mermaid

sequenceDiagram
    autonumber
    participant Client as "Клиент"
    participant System as "Система"
    participant Payment as "Платежный шлюз"
    participant Admin as "Администратор площадки"
    participant Contractor as "Подрядчик"
    participant Manager as "Менеджер"

    %% Запрос на бронирование
    Client->>System: Запрос доступности площадки
    alt Площадка доступна
        System-->>Client: Информация о стоимости и условиях
    else Площадка недоступна
        System-->>Client: Предложение выбрать другую дату/площадку
    end

    %% Подтверждение бронирования
    Client->>System: Подтверждает выбор площадки
    System->>Payment: Запрос предоплаты
    alt Платеж успешен
        Payment-->>System: Успешный платеж
        System-->>Client: Подтверждение бронирования
        System-->>Admin: Уведомление о бронировании
    else Платеж отклонен
        Payment-->>System: Отказ в оплате
        System-->>Client: Уведомление с предложением повторить оплату
    end

    %% Организация мероприятия
    Admin->>Contractor: Рассылка задач
    par Параллельное выполнение
        Contractor-->>System: Подтверждение выполнения задач
    end
    System-->>Admin: Отчет о выполнении задач

    %% Завершение мероприятия
    System-->>Client: Запрос оценки сервиса
    Client-->>System: Отправка отзывов
    System-->>Manager: Отчет по отзывам
